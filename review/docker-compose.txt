version: '3.8'

volumes:
  postgres_data:

services:
  db:
    image: postgres:latest
    volumes:
      - postgres_data:/var/lib/postgresql/data/
    env_file:
      - './.env'
  web:
    build: .
    restart: always
    command: gunicorn api_yamdb.wsgi:application --bind 0.0.0.0:8000
    volumes: 
      - ./static:/code/static
      - .:/code
    depends_on:
      - db
    env_file:
      - './.env'
    expose:
      - 8000
  nginx:
    build: ./nginx
    depends_on:
      - web
    ports:
      - 80:80
    volumes:
      - ./static:/home/app/static
